Quick Start

Install Dependencies

npm install ethers wagmi viem @rainbow-me/rainbowkit

Connect to KBT Token

import { ethers } from 'ethers';

const KBT_ADDRESS = "0x27533fEF27f07224e746705Da19243f89Dc36D81";
const KBT_ABI = [
  "function balanceOf(address) view returns (uint256)",
  "function transfer(address to, uint256 amount) returns (bool)",
  "function approve(address spender, uint256 amount) returns (bool)"
];

const provider = new ethers.providers.Web3Provider(window.ethereum);
const signer = provider.getSigner();
const kbtContract = new ethers.Contract(KBT_ADDRESS, KBT_ABI, signer);

// Get balance
const balance = await kbtContract.balanceOf(address);
console.log("KBT Balance:", ethers.utils.formatEther(balance));

Presale Contract

Buy with USDT

const PRESALE_ADDRESS = "0xF71e278443EBb18165a33Cf844bD9fDe8842B574";
const USDT_ADDRESS = "0xea70e6d9579110a965bEf06ff01d03216F5Ee1F1";

// 1. Approve USDT
const usdtContract = new ethers.Contract(USDT_ADDRESS, ERC20_ABI, signer);
const amount = ethers.utils.parseEther("100"); // 100 USDT
await usdtContract.approve(PRESALE_ADDRESS, amount);

// 2. Buy KBT
const presaleContract = new ethers.Contract(PRESALE_ADDRESS, PRESALE_ABI, signer);
const referrer = "0x0000000000000000000000000000000000000000"; // No referrer
await presaleContract.buyWithUSDT(amount, referrer);
Important: Always approve tokens before calling buy functions. The referrer address can be zero address if no referral.

Get Presale Info

const info = await presaleContract.getPresaleInfo();

console.log("Presale Start:", info.start);
console.log("Current Stage:", info.stage);
console.log("Tokens Sold:", ethers.utils.formatEther(info.tokensSold));
console.log("Stage Price:", ethers.utils.formatEther(info.stagePrice));

KBTPay Integration

Accept Payments

Integrate KBTPay to accept crypto payments in your application:

// Create payment intent
const response = await fetch('http://localhost:3000/api/transactions/create-intent', {
  method: 'POST',
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify({
    merchantAddress: "0xYourMerchantAddress",
    customerAddress: userAddress,
    amount: ethers.utils.parseEther("1.0").toString(),
    tokenAddress: "0x0000000000000000000000000000000000000000", // ETH
    chainId: 11155111
  })
});

const { paymentId } = await response.json();

// Send transaction
const tx = await signer.sendTransaction({
  to: merchantAddress,
  value: ethers.utils.parseEther("1.0")
});

// Update transaction hash
await fetch('http://localhost:3000/api/transactions/update-hash', {
  method: 'POST',
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify({
    paymentId: paymentId,
    transactionHash: tx.hash
  })
});

Web3 Integration

Using Wagmi (React)

import { useAccount, useWriteContract, usePublicClient } from 'wagmi';

function BuyKBT() {
  const { address } = useAccount();
  const { writeContractAsync } = useWriteContract();
  const publicClient = usePublicClient();

  async function buyTokens() {
    // Approve
    const approveTx = await writeContractAsync({
      address: USDT_ADDRESS,
      abi: ERC20_ABI,
      functionName: 'approve',
      args: [PRESALE_ADDRESS, amount]
    });
    await publicClient.waitForTransactionReceipt({ hash: approveTx });

    // Buy
    const buyTx = await writeContractAsync({
      address: PRESALE_ADDRESS,
      abi: PRESALE_ABI,
      functionName: 'buyWithUSDT',
      args: [amount, "0x0000000000000000000000000000000000000000"]
    });
    await publicClient.waitForTransactionReceipt({ hash: buyTx });
  }

  return ;
}
Tip: Use Wagmi hooks for better React integration and automatic state management.
